## 这一节解决什么问题

当 Builder 模式中某些步骤可能失败时，需要把方法链和错误处理结合：

```rust
// 普通 builder（所有步骤成功）
Connection::new()
    .host("localhost")
    .port(8080)
    .build()

// 带验证的 builder（某些步骤可能失败）
Connection::new()
    .host("localhost")?  // 验证主机名
    .port(-1)?           // ← 这里应该报错
    .build()
```

---

## 详细原理

### 方法返回 `Result<Self>` 让链式调用+错误处理共存

```rust
#[derive(Debug, Default)]
struct ConnectionBuilder {
    host: String,
    port: u16,
    timeout: u64,
}

impl ConnectionBuilder {
    fn new() -> Self { Self::default() }

    fn host(mut self, host: &str) -> Result<Self, String> {
        if host.is_empty() {
            return Err("主机名不能为空".into());
        }
        self.host = host.to_string();
        Ok(self)
    }

    fn port(mut self, port: i32) -> Result<Self, String> {
        if port <= 0 || port > 65535 {
            return Err(format!("无效端口: {port}"));
        }
        self.port = port as u16;
        Ok(self)
    }

    fn build(self) -> Result<String, String> {
        if self.host.is_empty() {
            return Err("必须设置主机名".into());
        }
        Ok(format!("{}:{}", self.host, self.port))
    }
}

fn create_connection(host: &str, port: i32) -> Result<String, String> {
    ConnectionBuilder::new()
        .host(host)?   // 失败立刻返回 Err
        .port(port)?   // 失败立刻返回 Err
        .build()       // 最终构建
}
```

## 完整运行示例

```rust
#[derive(Debug)]
struct Order {
    item: String,
    quantity: u32,
    price: f64,
}

impl Order {
    fn new() -> OrderBuilder {
        OrderBuilder::default()
    }
}

#[derive(Debug, Default)]
struct OrderBuilder {
    item: Option<String>,
    quantity: Option<u32>,
    price: Option<f64>,
}

impl OrderBuilder {
    fn item(mut self, item: &str) -> Result<Self, String> {
        if item.is_empty() { return Err("商品名不能为空".into()); }
        self.item = Some(item.to_string());
        Ok(self)
    }

    fn quantity(mut self, qty: i32) -> Result<Self, String> {
        if qty <= 0 { return Err(format!("数量必须大于0，得到: {qty}")); }
        self.quantity = Some(qty as u32);
        Ok(self)
    }

    fn price(mut self, price: f64) -> Result<Self, String> {
        if price < 0.0 { return Err(format!("价格不能为负: {price}")); }
        self.price = Some(price);
        Ok(self)
    }

    fn build(self) -> Result<Order, String> {
        Ok(Order {
            item: self.item.ok_or("缺少商品名")?,
            quantity: self.quantity.ok_or("缺少数量")?,
            price: self.price.ok_or("缺少价格")?,
        })
    }
}

fn create_order(item: &str, qty: i32, price: f64) -> Result<Order, String> {
    Order::new()
        .item(item)?
        .quantity(qty)?
        .price(price)?
        .build()
}

fn main() {
    let cases = [
        ("苹果", 5, 3.5),
        ("", 3, 2.0),      // 空名称
        ("香蕉", -1, 4.0), // 负数量
        ("橙子", 2, -1.0), // 负价格
    ];

    for (item, qty, price) in cases {
        match create_order(item, qty, price) {
            Ok(order) => println!("✅ 订单: {:?}", order),
            Err(e) => println!("❌ 创建失败: {e}"),
        }
    }
}
```
