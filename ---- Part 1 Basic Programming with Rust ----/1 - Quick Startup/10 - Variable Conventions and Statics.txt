// 			- Unused variables
// 			- Statics

fn main() {
    // Variable Convention and Unused Variables
    let _number_one = 45;
    let _x = 40_000;

    // Statics
    static WELCOME: &str = "Welcome to Rust";
    const PI: f32 = 3.14;

    let a = PI;
    let b = PI;

    let y = WELCOME;
    let z = WELCOME;
}
/*
--------------------------------------------------------------------------------------------------------
Topic                  | Summary
-----------------------|--------------------------------------------------------------------------------
Variable Convention    | Variables use snake_case, meaning all lowercase letters with underscores
                       | Unused variables start with an underscore
Numeric Readability    | Use underscores, which does not affect the value
Statics                | Declared with static and has an explicit type annotation
                       | Use SCREAMING_SNAKE_CASE; stored in a fixed memory location
---------------------------------------------------------------------------------------------------------
Statics vs Constants   | Constants are inlined with no memory address while statics occupy fixed memory
---------------------------------------------------------------------------------------------------------
*/
