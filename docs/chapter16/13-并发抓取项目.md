## 综合示例：并发处理 URL

把本章学到的技术结合起来：线程 + channel + 共享状态。

```rust
use std::sync::mpsc;
use std::thread;
use std::time::Duration;

// 模拟抓取（生产中换成真实 HTTP 请求）
fn fake_fetch(url: &str) -> Result<String, String> {
    thread::sleep(Duration::from_millis(100));
    if url.starts_with("http") {
        Ok(format!("<html>{url}</html>"))
    } else {
        Err(format!("无效 URL: {url}"))
    }
}

pub fn run() {
    let urls = vec!["http://a.com", "http://b.com", "bad_url", "http://c.com"];
    let (tx, rx) = mpsc::channel::<(&str, Result<String, String>)>();

    for url in &urls {
        let tx = tx.clone();
        let url = url; // 复制引用
        thread::spawn(move || {
            let result = fake_fetch(url);
            tx.send((url, result)).unwrap();
        });
    }
    drop(tx); // 原 tx drop，让 rx 能自然结束

    let (mut ok, mut err) = (0, 0);
    for (url, result) in rx {
        match result {
            Ok(content) => { ok += 1; println!("✅ {url}: {} bytes", content.len()); }
            Err(e) => { err += 1; println!("❌ {url}: {e}"); }
        }
    }

    println!("完成: {ok} 成功, {err} 失败");
}
```
