## 通过 `where Self: Sized` 让 trait 部分对象安全

```rust
trait Shape {
    fn area(&self) -> f64;    // 对象安全方法

    fn scale(self, factor: f64) -> Self  // ← 返回 Self，不对象安全
        where Self: Sized;               // 加这个就把它排除出 dyn

    fn clone_box(&self) -> Box<dyn Shape>; // 另一种方案：返回 Box
}

struct Circle { radius: f64 }

impl Shape for Circle {
    fn area(&self) -> f64 { std::f64::consts::PI * self.radius * self.radius }
    fn scale(self, factor: f64) -> Self where Self: Sized {
        Circle { radius: self.radius * factor }
    }
    fn clone_box(&self) -> Box<dyn Shape> {
        Box::new(Circle { radius: self.radius })
    }
}

let s: &dyn Shape = &Circle { radius: 5.0 };
println!("area = {:.2}", s.area()); // ✅
// s.scale(2.0); // ❌ dyn 不能调用 where Self: Sized 的方法
```
