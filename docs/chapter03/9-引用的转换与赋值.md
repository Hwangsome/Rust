## 这一节解决什么问题

引用在赋值时的行为和普通值不同：

```rust
let s = String::from("hello");
let r1 = &s;      // 借用
let r2 = r1;      // r2 也是对 s 的借用——r1 仍然有效！

let mut x = 5;
let m1 = &mut x;  // 可变借用
let m2 = m1;      // m2 接管了借用——m1 不再有效
```

`&T` 实现了 `Copy`（可以"复制引用"）；`&mut T` 不是 `Copy`（一次只能有一个可变引用）。

---

## 一分钟结论

- `&T`（不可变引用）实现了 `Copy`：赋值后两者都有效
- `&mut T`（可变引用）**不**实现 `Copy`：赋值是 move（原来的失效）
- **reborrow**：`&*mutable_ref` 从 `&mut T` 临时借一个 `&T` 出来
- **deref coercion**：`&String` 在需要 `&str` 时自动转换
- `==` 比较引用是比较**指向的值**；`ptr::eq` 比较**地址**

---

## 详细原理

### 1. `&T` 是 Copy

```rust
let s = String::from("hello");
let r1: &String = &s;
let r2 = r1;   // 复制引用，r1 和 r2 都有效

println!("{r1}");  // ✅
println!("{r2}");  // ✅
println!("{s}");   // ✅ s 没有被影响
```

### 2. `&mut T` 不是 Copy

```rust
let mut x = 5;
let r1: &mut i32 = &mut x;
let r2 = r1;  // r1 被 move 给 r2（&mut T 不是 Copy）

// println!("{}", r1); // ❌ r1 已经 move
*r2 += 1;
println!("x = {x}");  // 6
```

### 3. Reborrow

```rust
let mut s = String::from("hello");
let r_mut = &mut s;

// 从 &mut 里临时借一个 &（reborrow）
let r_read: &String = &*r_mut;  // 或者写 &r_mut
println!("{r_read}");  // r_read 的不可变借用结束

// 现在可以继续用 r_mut
r_mut.push_str(" world");
println!("{s}");
```

### 4. Deref Coercion

```rust
fn print_str(s: &str) { println!("{s}"); }
fn print_slice(s: &[i32]) { println!("{s:?}"); }

let owned = String::from("hello");
let boxed_string = Box::new(String::from("world"));
let v = vec![1, 2, 3];

// 自动 coercion 链
print_str(&owned);          // &String → &str
print_str(&boxed_string);   // &Box<String> → &String → &str
print_slice(&v);             // &Vec<i32> → &[i32]
print_slice(&[1, 2, 3]);    // &[i32; 3] → &[i32]
```

## 完整运行示例

```rust
use std::ptr;

fn main() {
    println!("=== &T 是 Copy ===");
    let data = vec![1, 2, 3, 4, 5];
    let r1 = &data;
    let r2 = r1;  // 复制引用

    println!("r1 = {r1:?}");
    println!("r2 = {r2:?}");
    println!("ptr::eq(r1, r2) = {}", ptr::eq(r1, r2));  // true：同一地址
    println!("data 仍然有效: {data:?}");
    println!();

    println!("=== &mut T 不是 Copy ===");
    let mut count = 0;
    let m1 = &mut count;
    *m1 += 10;
    let m2 = m1;  // move，不是 copy
    *m2 += 5;
    // println!("{}", m1); // ❌ m1 已经失效
    println!("count = {count}");  // 15
    println!();

    println!("=== Reborrow ===");
    let mut s = String::from("hello");
    let r_mut = &mut s;

    // 从可变引用里临时读取（reborrow）
    println!("reborrow 读取: {}", &*r_mut);
    // 读完后继续修改
    r_mut.push_str(" world");
    println!("修改后: {r_mut}");
    println!();

    println!("=== Deref Coercion ===");
    fn accept_str(s: &str) { println!("  str: {s}"); }
    fn accept_slice(s: &[i32]) { println!("  slice: {s:?}"); }

    accept_str("字面量");
    accept_str(&String::from("String"));
    accept_str(&Box::new(String::from("Box<String>")));

    let arr = [1, 2, 3];
    let v = vec![4, 5, 6];
    accept_slice(&arr);   // &[i32; 3] → &[i32]
    accept_slice(&v);     // &Vec<i32> → &[i32]
    println!();

    println!("=== 值相等 vs 地址相等 ===");
    let a = String::from("hello");
    let b = String::from("hello");
    let ra = &a;
    let rb = &b;
    let ra2 = ra;  // 复制引用

    println!("ra == rb: {} (值相等，内容相同)", ra == rb);
    println!("ptr::eq(ra, rb): {} (不同地址)", ptr::eq(ra, rb));
    println!("ptr::eq(ra, ra2): {} (相同地址，ra 和 ra2 指向同一个 a)", ptr::eq(ra, ra2));
}
```
