mod shapes {
    pub struct Circle {
        radius: f32,
    }
    impl Circle {
        pub fn new(radius: f32) -> Circle {
            Circle { radius }
        }

        pub fn new_1(radius: f32) -> Result<Circle, String> {
            if radius >= 0.0 {
                Ok(Circle { radius })
            } else {
                Err(String::from("radius should be positive"))
            }
        }

        pub fn new_2(radius: f32) -> Circle {
            match radius {
                -10.0..=0.0 => panic!("is between -10.0 and 0.0"),
                ..=-10.0 => panic!("is lesser then -10.0"),
                _ => Circle { radius },
            }
        }
        pub fn contains(&self, other: &Circle) -> bool {
            self.radius > other.radius
        }
    }
}

fn some_fn() {}
#[cfg(test)]
mod tests {
    use super::*;
    #[test]
    fn larger_circle_should_contain_smaller() {
        some_fn();
        let larger_circle = shapes::Circle::new(5.0);
        let smaller_circle = shapes::Circle::new(2.0);
        assert_eq!(
            larger_circle.contains(&smaller_circle),
            true,
            "Custom failure message"
        );

        assert_ne!(larger_circle.contains(&smaller_circle), false);
        assert!(larger_circle.contains(&smaller_circle));
    }

    #[test]
    fn smaller_circle_should_not_contain_larger() {
        let larger_circle = shapes::Circle::new(5.0);
        let smaller_circle = shapes::Circle::new(2.0);
        assert_eq!(!smaller_circle.contains(&larger_circle), true);
    }

    #[test]
    fn should_not_create_circle() -> Result<(), String> {
        let some_circle = shapes::Circle::new_1(-1.0)?;
        Ok(())
    }
    #[test]
    #[should_panic(expected = "is lesser then -10.0")]
    fn should_not_create_and_panic() {
        let some_circle = shapes::Circle::new_2(-11.0);
    }
}

// Summary of part 1
/*
------------------------------------------------------------------------------------------------
Concept/Topic            | Explanation
------------------------------------------------------------------------------------------------
Basics of Test Functions | - Test functions are typically placed in a separate modules of tests.
                         | - The tests module is annotated with configuration attributes
                         | - Test functions use the test annotation.

Typical Test Structure   | - Prepare the necessary inputs
                         | - Set up the initial conditions and invoke the function you want to test.
                         | - Verify results against the expected output using assertions.

Assertions and Panics    | - assert_eq!, assert_ne! and assert! macros validate expected results.
                         | - Failed assertions trigger a panic and test failure.
                         | - Custom failure messages can be added for clarity

Result in Test Functions | - Tests can return Result instead of panicking on failure.
                         | - The question mark operator helps propagate errors automatically.
------------------------------------------------------------------------------------------------
*/

// Summary of part 2
/*
----------------------------------------------------------------------------------------------------------
Concept/Topic                     |   Explanation
------------------------------------- --------------------------------------------------------------------
Testing Panics with should_panic | - Used to verify that a function panics under invalid conditions.
                                 | - Test fails if the function does not panic as expected.

Expected Panic Messages          | - should_panic can include an expected message check.
                                 | - Test passes only if the panic message contains the expected substring.

When to Use Result vs Panic      | - Result is for recoverable or user-facing errors.
                                 | - Panic is for unrecoverable errors where program cannot continue safely.

Additional Notes on Testing      | - Tests can be inside or outside the test module(standalone).
                                 | - Convention is that test function should be inside test module.
                                 | - Helper functions support test functions and have no test attribute.
                                 | - Tests can access private functions within the same module.
----------------------------------------------------------------------------------------------------------
 */
